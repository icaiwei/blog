defun(ArrayObjectAction (array_object_form)
    orix = axlFormGetField(array_object_form "orix")
    oriy = axlFormGetField(array_object_form "oriy")
    numx = axlFormGetField(array_object_form "numx")
    numy = axlFormGetField(array_object_form "numy")
    stepx = axlFormGetField(array_object_form "stepx")
    stepy = axlFormGetField(array_object_form "stepy")
    order = axlFormGetField(array_object_form "order")
    oddmi = axlFormGetField(array_object_form "oddmi")
    oddro = axlFormGetField(array_object_form "oddro")
    evenmi = axlFormGetField(array_object_form "evenmi")
    evenro = axlFormGetField(array_object_form "evenro")
    case(oddro
        ("0" oddro_value = 0);case
        ("1" oddro_value = 90);case
        ("2" oddro_value = 180);case
        ("3" oddro_value = 270);case
    );end-case
    case(evenro
        ("0" evenro_value = 0);case
        ("1" evenro_value = 90);case
        ("2" evenro_value = 180);case
        ("3" evenro_value = 270);case
    );end-case
    pin_filter_str = axlFormGetField(array_object_form "pinfilter")

    case(array_object_form->curField
        ;rectangle
        ("pickori"
            pt = axlEnterPoint()
            ptx = car(pt)
            pty = cadr(pt)
            axlFormSetField(array_object_form "orix" ptx)
            axlFormSetField(array_object_form "oriy" pty)
        );case
        ("placesym"
            axlFormListSelAll(array_object_form "selref" t)
            selected_ref_list = reverse(axlFormListGetSelItems(array_object_form "selref"))
            axlFormListSelAll(array_object_form "selref" nil)
            yushu = mod(length(selected_ref_list) numx * numy)
            if(yushu == 0 then
                n = 1
                for(i 1 numy
                    for(j 1 numx
                        axlClearSelSet()

defun(GetAngle (ori_ro cur_ro)
    case(cur_ro
        (0
            case(ori_ro
                (0.0 angle = 0 );case
                (90.0 angle = -90 );case
                (180.0 angle = -180 );case
                (270.0 angle = -270 );case
            );end-case
        );case
        (90
            case(ori_ro
                (0.0 angle = 90 );case
                (90.0 angle = 0 );case
                (180.0 angle = -90 );case
                (270.0 angle = -180 );case
            );end-case
        );case
        (180
            case(ori_ro

                (90.0 angle = 90 );case
                (180.0 angle = 0 );case
                (270.0 angle = -90 );case
            );end-case
        );case
        (270
            case(ori_ro
                (0.0 angle = 270 );case
                (90.0 angle = 180 );case
                (180.0 angle = 90 );case
                (270.0 angle = 0 );case
            );end-case
        );case
);end-let
);end-defun

;#                    遍历器件获取对应pin的net               #
;###########################################################
defun(GetPinNetListByRefList (ref_list pin_number)
    ;获取pin对应net
    pin_net_list = list()
    foreach(ref ref_list
        ref_dbid = car(axlGetSelSet(axlSelectByName("REFDES" ref)))
        foreach(pin_dbid ref_dbid->pins
            if(pin_dbid->number == pin_number then 
                pin_net_list = append1(pin_net_list pin_dbid->net->name)
            )
        );end-foreach
    );end-foreach
    pin_net_list = pin_net_list
);end-defun
