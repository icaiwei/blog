    Shape批量指定网络

    11/19/2021 Allegroskill

    # 说明

    微波板经常碰到很多铜皮，如果一个个指定网络需要花费很长时间，现二次开发出小程序，可批量指定网络

    # 源码

    axlCmdRegister("san" 'W_ShapeANet_Main)
procedure(W_ShapeANet_Main()
	CreateShapeANetForm()
	san_form=axlFormCreate( (gensym) san_form_file nil 'SanFormAction t)
	SanFormInit(san_form)
	axlFormDisplay(san_form)
	axlTempFileRemove(san_form_file)
)

defun(SanFormInit (san_form)
	axlClearSelSet()
	axlUIWPrint(san_form "Shape Assign Net Tool by wei.")
	;获取所有网络
	axlFormSetField(san_form "filter" "*")
	net_filter = axlFormGetField(san_form "filter")
	net_list = GetAllNets(net_filter)
	;设置所有网络到field
	axlFormSetField(san_form "allnet" net_list)
	
	axlFormColorize(san_form "info" 'text 'red)
	axlClearSelSet()
)

defun(SanFormAction (san_form)
	case(san_form->curField
		("select"
			axlClearSelSet()
			axlSetFindFilter(?enabled list("shapes") ?onButtons list("shapes"))
			;获取所选shape dbid
			shape_dbid_list = axlGetSelSet(axlAddSelectBox())
		)
		("allnet"
			selected_net_list = axlFormListGetSelItems(san_form "allnet")
			axlFormListAddItem(san_form "selnet" selected_net_list -1)
			axlFormListDeleteItem(san_form "allnet" selected_net_list)
		)
		("selnet"
			selected_net_list = axlFormListGetSelItems(san_form "selnet")
			axlFormListAddItem(san_form "allnet" selected_net_list -1)
			axlFormListDeleteItem(san_form "selnet" selected_net_list)
		)
		("filter"
			net_filter = axlFormGetField(san_form "filter")
			net_filter_list = GetAllNets(net_filter)
			
			axlFormListDeleteAll(san_form "allnet")
			axlFormSetField(san_form "allnet" net_filter_list)
		)
		("rightall"
			axlFormListDeleteAll(san_form "selnet")
			axlFormSetField(san_form "selnet" net_list)
			axlFormListDeleteAll(san_form "allnet")
		)
		("leftall"
			axlFormListDeleteAll(san_form "allnet")
			axlFormSetField(san_form "allnet" net_list)
			axlFormListDeleteAll(san_form "selnet")
		)
		("assign"
			shape_num = length(shape_dbid_list)
			axlFormListSelAll(san_form "selnet" t)
			selected_net_list_num = axlFormListGetSelCount(san_form "selnet")
			selected_net_list = axlFormListGetSelItems(san_form "selnet")
			if(shape_num == selected_net_list_num then
				for(i 1 shape_num
						axlChangeNet(nthelem(i shape_dbid_list) nthelem(i selected_net_list))
						axlUIWPrint(san_form "Shape assigning %s." nthelem(i selected_net_list))
				)
				axlUIWPrint(san_form "Shape assign completed.")
			else
				axlUIWPrint(san_form "Shape and net number does not match.")
			)
			axlFormListSelAll(san_form "selnet" nil)
			axlClearSelSet()
		)
		("close"
			axlFormClose(san_form)
			axlCancelEnterFun()
		)
		
	)
)

;获取所有网络名
defun(GetAllNets (filter_str)
	net_dbids = axlSelectByName("NET" filter_str t)
	net_list = nil
	foreach(dbid net_dbids
		net_list = append1(net_list dbid->name)
	)
	net_list = sort(net_list nil)
)
